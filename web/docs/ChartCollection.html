$.fn.ArrowFunctions = function( obj ) {

    var arrows = "<div class='up_arrow'></div>" +
        "<div class='down_arrow'></div>";

    return this.each( function() {

        $(this).append(arrows);
        $(this).find('.up_arrow, .down_arrow').unbind('click').bind('click', obj.arrow_click );
    });
};
